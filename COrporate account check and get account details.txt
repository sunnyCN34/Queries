  
 SELECT mp.MEMBER_NBR, I.FIRST_NAME, i.LAST_NAME,I.COMPANY_NAME,  s.SHARE_NBR, I.INDIVIDUAL_TYPE FROM MEMBERSHIPPARTICIPANT mp 
inner join INDIVIDUAL i on mp.INDIVIDUAL_ID = i.INDIVIDUAL_ID inner join SHARE s on mp.MEMBER_NBR = s.MEMBER_NBR
 AND I.INDIVIDUAL_TYPE != 3 order by mp.MEMBER_NBR desc     


SELECT FIRST_NAME,LAST_NAME,COMPANY_NAME, INDIVIDUAL_TYPE, CASE WHEN i.INDIVIDUAL_TYPE = 3 THEN i.COMPANY_NAME ELSE CONCAT(CONCAT(i.FIRST_NAME, ' '), i.LAST_NAME) END AS individualName 
                                   FROM DB2INST1.MEMBERSHIPPARTICIPANT mp
                                   INNER JOIN DB2INST1.INDIVIDUAL i ON mp.INDIVIDUAL_ID = i.INDIVIDUAL_ID AND mp.PARTICIPATION_TYPE = 101
                                   WHERE mp.MEMBER_NBR = 999999016
