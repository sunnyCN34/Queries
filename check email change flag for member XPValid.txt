SELECT COUNT(mp.MEMBER_NBR) FROM DATAAUDIT da INNER JOIN MEMBERSHIPPARTICIPANT mp ON mp.INDIVIDUAL_ID = REPLACE(substring(da.KEY_REPRESENTATION, 1, LOCATE('|', da.KEY_REPRESENTATION, 1) - 1), 'INDIVIDUAL_ID=', '')
AND mp.PARTICIPATION_TYPE = 101 WHERE da.TABLE_NAME = 'EMAIL' AND da.COLUMN_NAME = 'EMAIL_ADDRESS' AND da.ADDTIMESTAMP >=  '4/25/2018' AND da.ACTION_TYPE = 1 AND mp.MEMBER_NBR = 5883618
